# VCPE Debugging Guide Remote debugging with gdbserver and split debug files. PACKAGE_DEBUG_SPLIT_STYLE = "debug-with-srcpkg" ## Obtain rootfs and rootfs-dbg ```bash cd ~/git/meta-lxd/probes/tmp mkdir gdb tar xavf rootfs.tar.bz2 -C gdb tar xavf rootfs-dbg.tar.bz2 -C gdb ``` ## Start gdbserver in container ```bash lxc exec mv3-r21-9 bash gdbserver --attach $(pidof wanmanager) ``` ## Prepare gdbport ```bash lxc config device add mv3-r21-20 gdbport proxy listen=tcp:$192.168.2.120:2345 connect=tcp:127.0.0.1:2345 (lxc config device remove mv3-r21-20 gdbport) ``` ## Run gdb on host ```bash vi ~/.gdbinit set debug separate-debug-file on set pagination off gdb) set debug-file-directory usr/lib/debug gdb) set sysroot . gdb) file ./usr/bin/wanmanager gdb) target remote 192.168.2.120:2345 gdb) thread apply all bt full ``` ## core files ```bash root@mv2plus-r21-7:~$ ulimit -c unlimited root@mv2plus-r21-7:~$ cat /proc/sys/kernel/core_pattern root@mv2plus-r21-7:~$ kill -6 ```
